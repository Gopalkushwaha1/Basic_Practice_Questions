        return noofguess;
    }

        this.noofguess = noofguess;
    }

    void takeUserInput(){
        System.out.print("Guess the number :------>>>>>  ");
        Scanner sc = new Scanner(System.in);
        inputnumber = sc.nextInt();
    }

    boolean isCorrectNumber(){
        noOfGuesses++;
       if(inputnumber == number){
           System.out.format("Congratulations !!!! You guess the number : It was %d\n You guess in %d attemp " , number , noOfGuesses);
           return true;
       }
       else if (inputnumber < number ){
           System.out.println("OOPS!! Small number , *Think more* ");
       }
       else if(inputnumber > number){
           System.out.println("OOPS!!!! Large number , *Think less*");
        }
       return false;
    }
